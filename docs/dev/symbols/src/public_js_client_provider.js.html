1 define(['./framework', './lib/promise'], function (Framework, Promise) {
  2 
  3     /**
  4      * The ClientProvider module, renders the component's css, asks the framework to process the components
  5      * configuration, invokes the lifeCycle of the component and inserts the component's HTML in place.
  6      *
  7      * @name ClientProvider
  8      * @constructor
  9      */
 10     function ClientProvider () {}
 11 
 12     /**
 13      * Renders the components css and asks the framework to process the components configuration.
 14      * If any error occurs it renders an error message inside the page.
 15      *
 16      * @param {Object} configuration the component's configuration
 17      * @public
 18      */
 19     ClientProvider.prototype.render = function (configuration) {
 20         this._renderCss(configuration, configuration.css, configuration.path);
 21 
 22         var invokeLifeCycle = function(controller) {
 23             controller.init();
 24             controller.start();
 25         }
 26 
 27         $('#' + configuration.containerId).html(configuration.content);
 28         $('#' + configuration.containerId).css('visibility', 'hidden');
 29         configuration.cssLoaded.then(function () {
 30             console.log('aici', configuration);
 31             $('#' + configuration.containerId).css('visibility', 'visible');
 32         }, function (err) {
 33             //TODO: do something here
 34         });
 35         Framework.get().processComponent(configuration).then(function (controller) {
 36             console.log(configuration.content, configuration.containerId);
 37             if(controller) {
 38                 invokeLifeCycle(controller);
 39             }
 40         }, function (err) {
 41             //TODO: show an error
 42             //show an error box;
 43         });
 44     }
 45 
 46     /**
 47      * Renders the component's css in page.
 48      *
 49      * @param {Object} configuration the components configuration
 50      * @param {[String]} css an array of css paths for the current component
 51      * @param {String} location the component's location on the server
 52      * @private
 53      */
 54     ClientProvider.prototype._renderCss = function (configuration, css, location) {
 55         if (!css) {
 56             return;
 57         }
 58 
 59         var deferrers = [];
 60 
 61         var deferred = Promise.defer();
 62 
 63         var head = $("head"),
 64             linkElement;
 65 
 66         for(var i = 0, len = css.length; i < len; i++) {
 67             var cssPath = location + '/css/' + css[i],
 68                 sameLinks = head.find('link[href="' + cssPath + '"]');
 69 
 70 
 71             if (sameLinks.length === 0){
 72                 var deferrer = Promise.defer();
 73 
 74                 deferrers.push(deferrer);
 75                 linkElement = document.createElement('link');
 76                 linkElement.type = "text/css";
 77                 linkElement.rel = "stylesheet";
 78                 linkElement.href = cssPath;
 79                 linkElement.onload = function () {
 80                     deferrer.resolve();
 81                 }
 82                 head.append(linkElement);
 83             }
 84         }
 85 
 86         configuration.cssLoaded = deferred.promise;
 87         Promise.all(deferrer).then(function () {
 88             deferred.resolve();
 89         }, function (err) {
 90             deferred.reject(err);
 91         });
 92     }
 93 
 94     return ClientProvider;
 95 });
