 21 		currentDriver: null,

 24 		// Loads the contents of a text file from the local file system
 25 		// filePath is the path to the file in these formats:
 26 		//    x:\path\path\path\filename - PC local file
 27 		//    \\server\share\path\path\path\filename - PC network file
 28 		//    /path/path/path/filename - Mac/Unix local file
 29 		// returns the text of the file, or null if the operation cannot be performed or false if there was an error
 30 		load: function(filePath) {
 31 			var d = this.getDriver();
 32 			return d ? d.loadFile(filePath) : null;
 33 		},
 34 		// Saves a string to a text file on the local file system
 35 		// filePath is the path to the file in the format described above
 36 		// content is the string to save
 37 		// returns true if the file was saved successfully, or null if the operation cannot be performed or false if there was an error
 38 		save: function(filePath,content) {
 39 			var d = this.getDriver();
 40 			return d ? d.saveFile(filePath,content) : null;
 41 		},
 42 		// Copies a file on the local file system
 43 		// dest is the path to the destination file in the format described above
 44 		// source is the path to the source file in the format described above
 45 		// returns true if the file was copied successfully, or null if the operation cannot be performed or false if there was an error
 46 		copy: function(dest,source) {
 47 			var d = this.getDriver();
 48 			if(d && d.copyFile)
 49 				return d.copyFile(dest,source);
 50 			else
 51 				return null;
 52 		},
 53 		// Converts a local file path from the format returned by document.location into the format expected by this plugin
 54 		// url is the original URL of the file
 55 		// returns the equivalent local file path
 56 		convertUriToLocalPath: function (url) {
 57 			// Remove any location or query part of the URL
 58 			var originalPath = url.split("#")[0].split("?")[0];
 59 			// Convert file://localhost/ to file:///
 60 			if(originalPath.indexOf("file://localhost/") == 0)
 61 				originalPath = "file://" + originalPath.substr(16);
 62 			// Convert to a native file format
 63 			//# "file:///x:/path/path/path..." - pc local file --> "x:\path\path\path..."
 64 			//# "file://///server/share/path/path/path..." - FireFox pc network file --> "\\server\share\path\path\path..."
 65 			//# "file:///path/path/path..." - mac/unix local file --> "/path/path/path..."
 66 			//# "file://server/share/path/path/path..." - pc network file --> "\\server\share\path\path\path..."
 67 			var localPath;
 68 			if(originalPath.charAt(9) == ":") // PC local file
 69 				localPath = unescape(originalPath.substr(8)).replace(new RegExp("/","g"),"\\");
 70 			else if(originalPath.indexOf("file://///") == 0) // Firefox PC network file
 71 				localPath = "\\\\" + unescape(originalPath.substr(10)).replace(new RegExp("/","g"),"\\");
 72 			else if(originalPath.indexOf("file:///") == 0) // Mac/UNIX local file
 73 				localPath = unescape(originalPath.substr(7));
 74 			else if(originalPath.indexOf("file:/") == 0) // Mac/UNIX local file
 75 				localPath = unescape(originalPath.substr(5));
 76 			else if(originalPath.indexOf("//") == 0) // PC network file
 77 				localPath = "\\\\" + unescape(originalPath.substr(7)).replace(new RegExp("/","g"),"\\");
 78 			return localPath || originalPath;
 79 		},

 83 		// Returns a reference to the current driver
 84 		getDriver: function() {
 85 			if(this.currentDriver === null) {
 86 				for(var t=0; t<this.driverList.length; t++) {
 87 					if(this.currentDriver === null && drivers[this.driverList[t]].isAvailable && drivers[this.driverList[t]].isAvailable())
 88 						this.currentDriver = drivers[this.driverList[t]];
 89 				}
 90 			}
 91 			return this.currentDriver;
 92 		}
 93 	});

107 	// Internet Explorer driver
108 
109 	drivers.activeX = {
110 		name: "activeX",
111 		isAvailable: function() {
112 			try {
113 				var fso = new ActiveXObject("Scripting.FileSystemObject");
114 			} catch(ex) {
115 				return false;
116 			}
117 			return true;
118 		},
119 		loadFile: function(filePath) {
120 			// Returns null if it can't do it, false if there's an error, or a string of the content if successful
121 			try {
122 				var fso = new ActiveXObject("Scripting.FileSystemObject");
123 				var file = fso.OpenTextFile(filePath,1);
124 				var content = file.ReadAll();
125 				file.Close();
126 			} catch(ex) {
127 				//# alert("Exception while attempting to load\n\n" + ex.toString());
128 				return null;
129 			}
130 			return content;
131 		},
132 		createPath: function(path) {
133 			//# Remove the filename, if present. Use trailing slash (i.e. "foo\bar\") if no filename.
134 			var pos = path.lastIndexOf("\\");
135 			if(pos!=-1)
136 				path = path.substring(0,pos+1);
137 			//# Walk up the path until we find a folder that exists
138 			var scan = [path];
139 			try {
140 				var fso = new ActiveXObject("Scripting.FileSystemObject");
141 				var parent = fso.GetParentFolderName(path);
142 				while(parent && !fso.FolderExists(parent)) {
143 					scan.push(parent);
144 					parent = fso.GetParentFolderName(parent);
145 				}
146 				//# Walk back down the path, creating folders
147 				for(i=scan.length-1;i>=0;i--) {
148 					if(!fso.FolderExists(scan[i])) {
149 						fso.CreateFolder(scan[i]);
150 					}
151 				}
152 				return true;
153 			} catch(ex) {
154 			}
155 			return false;
156 		},
157 		copyFile: function(dest,source) {
158 			drivers.activeX.createPath(dest);
159 			try {
160 				var fso = new ActiveXObject("Scripting.FileSystemObject");
161 				fso.GetFile(source).Copy(dest);
162 			} catch(ex) {
163 				return false;
164 			}
165 			return true;
166 		},
167 		saveFile: function(filePath,content) {
168 			// Returns null if it can't do it, false if there's an error, true if it saved OK
169 			drivers.activeX.createPath(filePath);
170 			try {
171 				var fso = new ActiveXObject("Scripting.FileSystemObject");
172 				var file = fso.OpenTextFile(filePath,2,-1,0);
173 				file.Write(content);
174 				file.Close();
175 			} catch (ex) {
176 				return null;
177 			}
178 			return true;
179 		}
180 	};

182 	// Mozilla driver
183 
184 	drivers.mozilla = {
185 		name: "mozilla",
186 		isAvailable: function() {
187 			return !!window.Components;
188 		},
189 		loadFile: function(filePath) {
190 			// Returns null if it can't do it, false if there's an error, or a string of the content if successful
191 			if(window.Components) {
192 				try {
193 					netscape.security.PrivilegeManager.enablePrivilege("UniversalXPConnect");
194 					var file = Components.classes["@mozilla.org/file/local;1"].createInstance(Components.interfaces.nsILocalFile);
195 					file.initWithPath(filePath);
196 					if(!file.exists())
197 						return null;
198 					var inputStream = Components.classes["@mozilla.org/network/file-input-stream;1"].createInstance(Components.interfaces.nsIFileInputStream);
199 					inputStream.init(file,0x01,00004,null);
200 					var sInputStream = Components.classes["@mozilla.org/scriptableinputstream;1"].createInstance(Components.interfaces.nsIScriptableInputStream);
201 					sInputStream.init(inputStream);
202 					var contents = sInputStream.read(sInputStream.available());
203 					sInputStream.close();
204 					inputStream.close();
205 					return contents;
206 				} catch(ex) {
207 					//# alert("Exception while attempting to load\n\n" + ex);
208 					return false;
209 				}
210 			}
211 			return null;
212 		},
213 		saveFile: function(filePath,content) {
214 			// Returns null if it can't do it, false if there's an error, true if it saved OK
215 			if(window.Components) {
216 				try {
217 					netscape.security.PrivilegeManager.enablePrivilege("UniversalXPConnect");
218 					var file = Components.classes["@mozilla.org/file/local;1"].createInstance(Components.interfaces.nsILocalFile);
219 					file.initWithPath(filePath);
220 					if(!file.exists())
221 						file.create(0,0664);
222 					var out = Components.classes["@mozilla.org/network/file-output-stream;1"].createInstance(Components.interfaces.nsIFileOutputStream);
223 					out.init(file,0x20|0x02,00004,null);
224 					out.write(content,content.length);
225 					out.flush();
226 					out.close();
227 					return true;
228 				} catch(ex) {
229 					alert("Exception while attempting to save\n\n" + ex);
230 					return false;
231 				}
232 			}
233 			return null;
234 		}

239 	drivers.tiddlySaver = {
240 		name: "tiddlySaver",
241 		deferredInit: function() {
242 			if(!document.applets["TiddlySaver"] && !$.browser.mozilla && !$.browser.msie && document.location.toString().substr(0,5) == "file:") {
243 				$(document.body).append("<applet style='position:absolute;left:-1px' name='TiddlySaver' code='TiddlySaver.class' archive='TiddlySaver.jar' width='1'height='1'></applet>");
244 			}
245 		},
246 		isAvailable: function() {
247 			return !!document.applets["TiddlySaver"];
248 		},
249 		loadFile: function(filePath) {
250 			var r;
251 			try {
252 				if(document.applets["TiddlySaver"]) {
253 					r = document.applets["TiddlySaver"].loadFile(javaUrlToFilename(filePath),"UTF-8");
254 					return (r === undefined || r === null) ? null : String(r);
255 				}
256 			} catch(ex) {
257 			}
258 			return null;
259 		},
260 		saveFile: function(filePath,content) {
261 			try {
262 				if(document.applets["TiddlySaver"])
263 					return document.applets["TiddlySaver"].saveFile(javaUrlToFilename(filePath),"UTF-8",content);
264 			} catch(ex) {
265 			}
266 			return null;
267 		}
268 	}

272 	drivers.javaLiveConnect = {
273 		name: "javaLiveConnect",
274 		isAvailable: function() {
275 			return !!window.java && !!window.java.io && !!window.java.io.FileReader;
276 		},
277 		loadFile: function(filePath) {
278 			var r;
279 			var content = [];
280 			try {
281 				r = new java.io.BufferedReader(new java.io.FileReader(javaUrlToFilename(filePath)));
282 				var line;
283 				while((line = r.readLine()) != null)
284 					content.push(new String(line));
285 				r.close();
286 			} catch(ex) {
287 				return null;
288 			}
289 			return content.join("\n") + "\n";
290 		},
291 		saveFile: function(filePath,content) {
292 			try {
293 				var s = new java.io.PrintStream(new java.io.FileOutputStream(javaUrlToFilename(filePath)));
294 				s.print(content);
295 				s.close();
296 			} catch(ex) {
297 				return null;
298 			}
299 			return true;
300 		}
301 	}

303 	// Private utilities

305 	function javaUrlToFilename(url) {
306 		var f = "//localhost";
307 		if(url.indexOf(f) == 0)
308 				return url.substring(f.length);
309 		var i = url.indexOf(":");
310 		return i > 0 ? url.substring(i-1) : url;
311 	}
