156 function Deferred(canceller){
157 	var result, finished, isError, waiting = [], handled;
158 	var promise = this.promise = new Promise();
159 	var context = exports.currentContext;
160 	
161 	function notifyAll(value){
162 		var previousContext = exports.currentContext;
163 		if(finished){
164 			throw new Error("This deferred has already been resolved");				
165 		}
166 		try{
167 			if(previousContext !== context){
168 				if(previousContext && previousContext.suspend){
169 					previousContext.suspend();
170 				}
171 				exports.currentContext = context;
172 				if(context && context.resume){
173 					context.resume();
174 				}
175 			}
176 			result = value;
177 			finished = true;
178 			for(var i = 0; i < waiting.length; i++){
179 				notify(waiting[i]);	
180 			}
181 		}
182 		finally{
183 			if(previousContext !== context){
184 				if(context && context.suspend){
185 					context.suspend();
186 				}
187 				if(previousContext && previousContext.resume){
188 					previousContext.resume();
189 				}
190 				exports.currentContext = previousContext;
191 			}
192 		}
193 	}
194 	function notify(listener){
195 		var func = (isError ? listener.error : listener.resolved);
196 		if(func){
197 			handled = true;
198 				try{
199 					var newResult = func(result);
200 					if(newResult && typeof newResult.then === "function"){
201 						newResult.then(listener.deferred.resolve, listener.deferred.reject);
202 						return;
203 					}
204 					listener.deferred.resolve(newResult);
205 				}
206 				catch(e){
207 					listener.deferred.reject(e);
208 				}
209 		}
210 		else{
211 			if(isError){
212 				if (listener.deferred.reject(result, true)) {
213 					handled = true;
214 					}
215 			}
216 			else{
217 				listener.deferred.resolve.call(listener.deferred, result);
218 			}
219 		}
220 	}
221 	// calling resolve will resolve the promise
222 	this.resolve = this.callback = this.emitSuccess = function(value){
223 		notifyAll(value);
224 	};
225 	
226 	// calling error will indicate that the promise failed
227 	var reject = this.reject = this.errback = this.emitError = function(error, dontThrow){
228 		isError = true;
229 		notifyAll(error);
230 		if (!dontThrow && typeof setTimeout !== "undefined") {
231 			setTimeout(function () {
232 				if (!handled) {
233 					throw error;
234 				}
235 			}, exports.errorTimeout);
236 		}
237 		return handled;
238 	};
239 	
240 	// call progress to provide updates on the progress on the completion of the promise
241 	this.progress = function(update){
242 		for(var i = 0; i < waiting.length; i++){
243 			var progress = waiting[i].progress;
244 			progress && progress(update);	
245 		}
246 	}
247 	// provide the implementation of the promise
248 	this.then = promise.then = function(resolvedCallback, errorCallback, progressCallback){
249 		var returnDeferred = new Deferred(promise.cancel);
250 		var listener = {resolved: resolvedCallback, error: errorCallback, progress: progressCallback, deferred: returnDeferred}; 
251 		if(finished){
252 			notify(listener);
253 		}
254 		else{
255 			waiting.push(listener);
256 		}
257 		return returnDeferred.promise;
258 	};
259 	var timeout;
260 	if(typeof setTimeout !== "undefined") {
261 		this.timeout = function (ms) {
262 			if (ms === undefined) {
263 				return timeout;
264 			}
265 			timeout = ms;
266 			setTimeout(function () {
267 				if (!finished) {
268 					if (promise.cancel) {
269 						promise.cancel(new Error("timeout"));
270 					}
271 					else {
272 						reject(new Error("timeout"));
273 					}
274 				}
275 			}, ms);
276 			return promise;
277 		};
278 	}
279 	
280 	if(canceller){
281 		this.cancel = promise.cancel = function(){
282 			var error = canceller();
283 			if(!(error instanceof Error)){
284 				error = new Error(error);
285 			}
286 			reject(error);
287 		}
288 	}
289 	freeze(promise);
290 };

292 function perform(value, async, sync){
293 	try{
294 		if(value && typeof value.then === "function"){
295 			value = async(value);
296 		}
297 		else{
298 			value = sync(value);
299 		}
300 		if(value && typeof value.then === "function"){
301 			return value;
302 		}
303 		var deferred = new Deferred();
304 		deferred.resolve(value);
305 		return deferred.promise;
306 	}catch(e){
307 		var deferred = new Deferred();
308 		deferred.reject(e);
309 		return deferred.promise;
310 	}
311 	
312 }

319 /**
320  * Registers an observer on a promise, always returning a promise
321  * @param value		 promise or value to observe
322  * @param resolvedCallback function to be called with the resolved value
323  * @param rejectCallback	function to be called with the rejection reason
324  * @param progressCallback	function to be called when progress is made
325  * @return promise for the return value from the invoked callback
326  */
327 exports.whenPromise = function(value, resolvedCallback, rejectCallback, progressCallback){
328 	var deferred = defer();
329 	if(value && typeof value.then === "function"){
330 		value.then(function(next){
331 			deferred.resolve(next);
332 		},function(error){
333 			deferred.reject(error);
334 		});
335 		rejectCallback = rejectCallback || rethrow;
336 	}else{
337 		deferred.resolve(value);
338 	}
339 	return deferred.promise.then(resolvedCallback, rejectCallback, progressCallback);
340 };

342 /**
343  * Registers an observer on a promise.
344  * @param value		 promise or value to observe
345  * @param resolvedCallback function to be called with the resolved value
346  * @param rejectCallback	function to be called with the rejection reason
347  * @param progressCallback	function to be called when progress is made
348  * @return promise for the return value from the invoked callback or the value if it
349  * is a non-promise value
350  */
351 exports.when = function(value, resolvedCallback, rejectCallback, progressCallback){
352 		if(value && typeof value.then === "function"){
353 				if(value instanceof Promise){
354 						return value.then(resolvedCallback, rejectCallback, progressCallback);
355 				}
356 				else{
357 						return exports.whenPromise(value, resolvedCallback, rejectCallback, progressCallback);
358 				}
359 		}
360 		return resolvedCallback ? resolvedCallback(value) : value;
361 };

427  * Waits for the given promise to finish, blocking (and executing other events)
428  * if necessary to wait for the promise to finish. If target is not a promise
429  * it will return the target immediately. If the promise results in an reject,
430  * that reject will be thrown.
431  * @param target	 promise or value to wait for.
432  * @return the value of the promise;
433  */
434 var queue;
435 //try {
436 //		queue = require("event-loop");
437 //}
438 //catch (e) {}
439 exports.wait = function(target){
440 	if(!queue){
441 		throw new Error("Can not wait, the event-queue module is not available");
442 	}
443 	if(target && typeof target.then === "function"){
444 		var isFinished, isError, result;		
445 		target.then(function(value){
446 			isFinished = true;
447 			result = value;
448 		},
449 		function(error){
450 			isFinished = true;
451 			isError = true;
452 			result = error;
453 		});
454 		while(!isFinished){
455 			queue.processNextEvent(true);
456 		}
457 		if(isError){
458 			throw result;
459 		}
460 		return result;
461 	}
462 	else{
463 		return target;
464 	}

470  * Takes an array of promises and returns a promise that is fulfilled once all
471  * the promises in the array are fulfilled
472  * @param array	The array of promises
473  * @return the promise that is fulfilled when all the array is fulfilled, resolved to the array of results
474  */
475 exports.all = function(array){
476 	var deferred = new Deferred();
477 	if(!(array instanceof Array)){
478 		array = Array.prototype.slice.call(arguments);
479 	}
480 	var fulfilled = 0, length = array.length;
481 	var results = [];
482 	if (length === 0) deferred.resolve(results);
483 	else {	
484 		array.forEach(function(promise, index){
485 			exports.when(promise, 
486 				function(value){
487 					results[index] = value;
488 					fulfilled++;
489 					if(fulfilled === length){
490 						deferred.resolve(results);
491 					}
492 				},
493 				deferred.reject);
494 		});
495 	}
496 	return deferred.promise;

601  * @param func	 node compatible async function which takes a callback as its last argument

603  */

605 	var args = Array.prototype.slice.call(arguments, 1);
606 
607 	var deferred = new Deferred();
608 	args.push(function(error, result){
609 		if(error) {
610 			deferred.emitError(error);
611 		}
612 		else {
613 			if(arguments.length > 2){
614 				// if there are multiple success values, we return an array
615 				Array.prototype.shift.call(arguments, 1);
616 				deferred.emitSuccess(arguments);
617 			}
618 			else{
619 				deferred.emitSuccess(result);
620 			}
621 		}
622 	});
623 	asyncFunction.apply(this, args);
624 	return deferred.promise;
