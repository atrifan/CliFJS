1 define([], function () {
  2     /**
  3      * The client's component map.
  4      *
  5      * @name ComponentMap
  6      * @constructor
  7      */
  8     function ComponentMap () {
  9         /**
 10          * The componentMap property.
 11          *
 12          * @type {Object}
 13          * @private
 14          */
 15         this._componentMap = {};
 16     };
 17 
 18     /**
 19      * The instance of the ComponentMap module.
 20      *
 21      * @type {ComponentMap}
 22      * @private
 23      */
 24     ComponentMap._instance = null;
 25 
 26     /**
 27      * Singleton get method for the ComponentMap instance.
 28      *
 29      * @returns {ComponentMap}
 30      * @static
 31      */
 32     ComponentMap.get = function () {
 33         return ComponentMap._instance ||
 34             (ComponentMap._instance = new ComponentMap());
 35     }
 36 
 37     /**
 38      * Gets a component with a specific id.
 39      *
 40      * @param {String} id the id of the component
 41      * @returns {Object} the found component in the componentMap.
 42      * @public
 43      */
 44     ComponentMap.prototype.getComponent = function (id) {
 45         return this._componentMap[id];
 46     };
 47 
 48     /**
 49      * Gets the component map.
 50      *
 51      * @returns {Object} the client's component map.
 52      * @public
 53      */
 54     ComponentMap.prototype.getComponentMap = function () {
 55         return this._componentMap;
 56     };
 57 
 58     /**
 59      * Registers a component to the client's component map.
 60      *
 61      * @param {String} id the unique id of the component's entrance
 62      * @param {Object} configuration the configuration that you want to register for a component
 63      * @public
 64      */
 65     ComponentMap.prototype.registerComponent = function (id, configuration) {
 66         this._componentMap[id] = configuration;
 67     };
 68 
 69     /**
 70      * Removes a component with a specific id from the componentMap.
 71      *
 72      * @param {String} id the id for the component that has to be removed
 73      */
 74     ComponentMap.prototype.removeComponent = function (id) {
 75         delete this._componentMap[id];
 76     };
 77 
 78     /**
 79      * Empties the client componentMap.
 80      *
 81      * @public
 82      */
 83     ComponentMap.prototype.flushComponents = function () {
 84         delete this._componentMap;
 85 
 86         this._componentMap = {};
 87     }
 88 
 89 
 90     return ComponentMap;
 91 });
