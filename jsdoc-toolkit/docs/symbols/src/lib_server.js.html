  6 	session = require('./middleware_handlers/session'),
  7     logging = require('./logging/logger'),

 28 var requestHandler = require('./middleware_handlers/requestHandler');
 29 function Server() {}
 30 
 31 Server.prototype.start = function () {
 32 	this.configuration = configuration;
 33 	var sessionStore = factoryStore.get();
 34 	this.httpServer = connect()
 35 	 					.use(connect.favicon('icon/DEARC2.ico'))
 36 	 					.use(connect.cookieParser(this.configuration.secret))
 37 	 					.use(connect.query())
 38 	 					.use(connect.bodyParser())
 39 						.use(requestHandler())
 40 						.use(session.get(
 41 								{store: sessionStore}))
 42                         .use(function callRouteHandler(request, response, next) {
 43                             var route = request.applicationRoute;
 44                             if (route) {
 45                                 delete request.applicationRoute;
 46                                 route.handle(request, response);
 47                             } else {
 48                                 next();
 49                             }
 50                         })
 51 						.use(function (err, req, res, next) {
 52                                 logger.error(err.stacktrace);
 53 								throw new Error(err, err.stacktrace);
 54 								return;
 55 						})
 56 						.listen(this.configuration.port);
 57      					
 58 	this.socket = io.listen(this.httpServer, {
 59         logger: logging.get('socketIO')
 60     });
 61 
 62     this.socket.set('transports', ['websocket', 'xhr-polling', 'jsonp-polling']);
 63 }
 64 
 65 Server.prototype.stop = function () {
 66 	 try {
 67 	        var clients = this.socket.sockets.clients();
 68 
 69 	        for (var i = clients.length; i--;) {
 70 	            clients[i].disconnect();
 71 	        }
 72 	        this.httpServer.close();
 73 	 } catch (err) {
 74 		 throw new Error('Unable to shutdown');
 75 	 }
 76 }
 77 
 78 module.exports = new Server();
