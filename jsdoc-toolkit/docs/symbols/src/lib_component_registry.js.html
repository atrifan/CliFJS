  9 function ComponentRegistry () {
 10     this._componentMap = {};
 11     if(config.components) {
 12         for(var i = 0, len = config.components.length; i < len; i++) {
 13             this._walkComponents(config.components[i]);
 14         }
 15     } else {
 16         this._walkComponents(path.join(process.cwd(), './components'));
 17     }
 18 
 19     //this._registerClientCore();
 20 };

 33 ComponentRegistry.prototype._walkComponents = function (path) {
 34     try {
 35         var components = fs.readdirSync(path);
 36         for (var i = 0, len = components.length; i < len; i++) {
 37             try {
 38                 var componentLocation = path + '\\' + components[i];
 39                 var componentConfiguration = require(componentLocation + '\\meta.json');
 40                 this._registerComponent(componentConfiguration, componentLocation);
 41                 logger.debug(util.format('Registered component %s', components[i]));
 42             } catch (err) {
 43                 logger.warn(util.format('Could not load %s\\%s\'s cofiguration', path, components[i]), err);
 44             }
 45         }
 46     } catch (err) {
 47         logger.error(util.format('Could not read %s', path), err);
 48         throw new Error(err);
 49     }
 50 };
 51 
 52 ComponentRegistry.prototype._registerComponent = function(configuration, location) {
 53     if (!configuration.id || !(/^(\d+\.)?(\d+\.)(\d+)$/.test(configuration.version))) {
 54         logger.warn(util.format('Component %s could not be registered', configuration.id));
 55         return;
 56     }
 57 
 58     var id = configuration.id,
 59         version = configuration.version;
 60 
 61     configuration.location = location;
 62     configuration.useSession = configuration.useSession || false;
 63 
 64     if (!this._componentMap[id]) {
 65         this._componentMap[id] = {
 66             configuration: {},
 67             versions: []
 68         };
 69     }
 70 
 71     var componentsEntry = this._componentMap[id];
 72 
 73     // prevent overriding an already registered component
 74     if (typeof componentsEntry.configuration[version] !== 'undefined') {
 75         logger.warn(util.format('For component %s, the version %s was deployed multiple times',
 76             id, version));
 77         return;
 78     }
 79 
 80     componentsEntry.configuration[version] = configuration;
 81 
 82     componentsEntry.versions.push(version);
 83 
 84     for(var i = 0, len = componentsEntry.versions.length; i < len - 1; i++) {
 85         for(var j = i + 1; j < len; j++) {
 86             if(componentsEntry.versions[i] > componentsEntry.versions[j]) {
 87                 var aux = componentsEntry.versions[i];
 88                 componentsEntry.versions[i] = componentsEntry.versions[j];
 89                 componentsEntry.versions[j] = aux;
 90             }
 91         }
 92     }
 93 };
 94 
 95 ComponentRegistry.prototype.getConfiguration = function(id, version) {
 96     if (!this._componentMap[id] || !this._componentMap[id].configuration[version]) {
 97         logger.warn(util.format('Couldn\'t get configuration for: %s;%s.', id, version));
 98         return;
 99     }
100 
101     return this._componentMap[id].configuration[version];
102 };
