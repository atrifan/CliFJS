 25     }

 27     return;
 28 };
 29 
 30 BaseSession.prototype.remove = function (key) {
 31 
 32     if(typeof this._session[key] === 'undefined') {
 33         return;
 34     }
 35 
 36     if(this._updatedKeys.indexOf(key) !== -1) {
 37         this._updatedKeys.splice(this._updatedKeys.indexOf(key), 1);
 38     }

 41     this._removedKeys.push(key);
 42 };
 43 
 44 BaseSession.prototype.removeAll = function () {

 51 };
