1 "use strict";
  2 
  3 var path = require('path'),
  4     fs = require('fs'),
  5     componentRegistry = require('./component_registry'),
  6     logger = require('./logging/logger').get(),
  7     handlebars = require('handlebars');
  8 
  9 function PrecompileTemplates() {
 10     var componentMap = componentRegistry._componentMap
 11     for(var component in componentMap) {
 12         for (var version in componentMap[component].configuration) {
 13             for(var view in componentMap[component].configuration[version].views) {
 14                 var location = componentMap[component].configuration[version].location,
 15                     htmlRoute = componentMap[component].configuration[version].views[view].view;
 16                 this._precompileTemplate(location, htmlRoute, componentMap[component].configuration[version].views[view]);
 17             }
 18         }
 19     }
 20 }
 21 
 22 PrecompileTemplates.prototype._precompileTemplate = function(location, htmlRoute, component) {
 23     var locationOnDisk = path.join(location, '/client/html/', htmlRoute);
 24     try {
 25         var templateData = fs.readFileSync(locationOnDisk);
 26         templateData = templateData.toString();
 27         var compilation = handlebars.compile(templateData);
 28         component.content = compilation;
 29 
 30     } catch (err) {
 31         throw new Error(err.stack);
 32     }
 33 }
 34 
 35 module.exports = new PrecompileTemplates();
