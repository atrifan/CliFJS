1 "use strict";
  2 
  3 var fs = require('fs'),
  4     path = require('path'),
  5     componentRegistry = require('../component_registry'),
  6     serverRenderer = require('../server_renderer');
  7 
  8 function ViewHandler() {}
  9 
 10 ViewHandler.prototype.match = function (request) {
 11     //aici ar trebuiii sa se seteze useSession True sau false. sau pe renderer.
 12     var requestPath = request._parsedUrl.pathname,
 13         splitPath = requestPath.split('/').slice(1);
 14 
 15 
 16     if(splitPath.length >= 1 && splitPath.length <= 3 &&
 17         splitPath.indexOf('css') === -1 && splitPath.indexOf('js') === -1 &&
 18         path.extname(requestPath) !== '.js' && path.extname(requestPath) !== '.css') {
 19         request.component = {
 20             id: splitPath[0],
 21             version: splitPath.length === 3 ? splitPath[1] : null, //on else get latest version
 22             view: (splitPath[1] && splitPath.length === 2) ? splitPath[1] :
 23                 ((splitPath.length === 3 && splitPath[2]) ? splitPath[2] : 'index')
 24         }
 25         return true;
 26     } else {
 27         return false;
 28     }
 29 };
 30 
 31 ViewHandler.prototype.handle = function (request, response) {
 32     response.setHeader('Content-Type', 'text/html; charset=UTF-8');
 33     response.setHeader('Cache-Control', 'no-cache, must-revalidate');
 34     response.setHeader('Pragma', 'no-cache');
 35     response.setHeader('Expires', new Date(Date.now() - 1000 * 60 * 60).toUTCString());
 36 
 37     var componentId = request.component.id,
 38         componentVersion = request.component.version,
 39         componentView = request.component.view;
 40 
 41     /*verify if the component requested really exists in my map*/
 42     if (componentRegistry._componentMap[componentId] &&
 43         componentRegistry._componentMap[componentId].versions.indexOf(componentVersion) !== -1 &&
 44         componentRegistry._componentMap[componentId].configuration[componentVersion].views[componentView])  {
 45 
 46         var requestedComponent = componentRegistry._componentMap[componentId].configuration[componentVersion];
 47         var requestedView = requestedComponent.views[componentView]
 48 
 49         //pass the requestedComponent to the server renderer
 50         if(!request.query.agregatte) {
 51             var coreComponent = componentRegistry._componentMap['core'].configuration['1.0'];
 52             serverRenderer.renderCore(coreComponent, 'index', requestedComponent, componentView, response);
 53         } else {
 54             serverRenderer.render(requestedComponent, componentView, response);
 55         }
 56     } else {
 57         response.writeHeader(404, {'Content-Type': 'text/plain'});
 58         response.end("Server error: Could not find " + request.url);
 59     }
 60 
 61 };
 62 
 63 module.exports = new ViewHandler();
