1 "use strict";
  2 
  3 var crypto = require('crypto'),
  4     config = require('./../configuration'),
  5     utils = require('connect/lib/utils'),
  6     Promise = require('promised-io/promise'),
  7     Cookie = require('connect/lib/middleware/session/cookie'),
  8     signature = require('connect/node_modules/cookie-signature'),
  9     util = require('util'),
 10     logger = require('./../logging/logger').get();
 11 
 12 var SESSION_REQUEST_ROUTE = ['controller', 'view'];
 13 
 14 /**
 15  * The Session middleware module. Sets a session on the request.
 16  *
 17  * @param {Object} sessionConf the session configuration for instantiating the Session middleware handler
 18  * @name Session
 19  * @constructor
 20  */
 21 function Session (sessionConf) {
 22 	this._store = sessionConf.store;
 23     this._key = sessionConf.key || 'DEARC.on';
 24     this._cookie = sessionConf.cookie || {};
 25 };
 26 
 27 /**
 28  * The instance of the Session middleware module.
 29  * @type {Session}
 30  * @private
 31  */
 32 Session._instance = null
 33 
 34 /**
 35  * Singleton get method.
 36  *
 37  * @param {Object} configuration the configuration needed to instantiate the Session module.
 38  * @returns {Session} the Session middleware handler instance
 39  */
 40 Session.get = function (configuration) {
 41     if (!Session._instance) {
 42         Session._instance = new Session(configuration);
 43     }
 44 
 45     return Session._instance.handle.bind(Session._instance);
 46 };
 47 
 48 /**
 49  * Handles the HTTP request and sets a session for that request. It sets a cookie on the request if none is set
 50  * it gets or creates a session for an existing cookie. It saves the session on the end of every request.
 51  *
 52  * @param {Request} request the HTTP request Object.
 53  * @param {Response} response the HTTP response Object.
 54  * @param {Function} next the connect.middleware next function.
 55  */
 56 Session.prototype.handle = function(request, response, next) {
 57 	var self = this;
 58 	
 59 	//you should set the cookie only once, depending on a session store see cookie from connect/lib/middleware/session/cookie
 60 	//cookie will be set for verification if get sessionId... figure it out
 61 	if (request.globalSession || !this._needsSession(request)) {
 62 		next();
 63         return;
 64 	}
 65 
 66 	request.sessionStore = self._store;
 67 
 68     request.sessionId = request.signedCookies[this._key];
 69 
 70     var pause = utils.pause(request);
 71 
 72     this._provisionWithSession(request).then(
 73         function () {
 74             next();
 75             pause.resume();
 76         },
 77         function (err) {
 78             next(err);
 79             pause.resume();
 80         }
 81     );
 82 
 83     response.on('header', function () {
 84         self._setCookie(request, response);
 85     });
 86 
 87     this._saveSessionOnEnd(request, response);
 88 	
 89 };
 90 
 91 /**
 92  * Verifies if a request needs session on route depending on the ``applicationRoute.type`` property of the request,
 93  * set on the request handler middleware.
 94  *
 95  * @param {Request} request the HTTP request object.
 96  * @returns {boolean} true or false.
 97  * @private
 98  */
 99 Session.prototype._needsSession = function (request) {
100     if (request.applicationRoute){
101         return SESSION_REQUEST_ROUTE.indexOf(request.applicationRoute.type) !== -1;
102     }
103 };
104 
105 /**
106  * Provisiones the request with a session object. If the request has already a sessionId than it gets
107  * the session from the session store, otherwise it creates a new SessionId for the request and creates a new
108  * entry in the session store for the generated unique id.
109  *
110  * @param {Request} request the HTTP request Object.
111  * @returns {Object} the session Object to be set on the request.
112  * @private
113  */
114 Session.prototype._provisionWithSession = function (request) {
115     var deferred = Promise.defer(),
116         self = this;
117 
118     if(!request.sessionId) {
119         request.sessionId = this._generateSessionId();
120         request.sessionIsNew = true;
121 
122         this._store.createNewSession(request.sessionId).then(function (session) {
123             request.globalSession = session;
124             logger.debug(util.format('Created a new session: %s', request.sessionId));
125             deferred.resolve();
126         }, function (err) {
127             logger.error(util.format('Failed to create a new session: %s', request.sessionId),
128                 err);
129             deferred.reject(err);
130         });
131     } else {
132         this._store.get(request.sessionId).then(function (session) {
133             // generate a new session id if a session doesn't exist for the id passed by user
134             if (session.isEmpty()) {
135                 logger.debug(util.format('A session doesn\'t exist for: %s', request.sessionId));
136                 request.sessionId = null;
137                 self._provisionWithSession(request).then(deferred.resolve, deferred.reject);
138                 return;
139             }
140 
141             request.globalSession = session;
142             logger.debug(util.format('Retrieved the global session for: %s', request.sessionId));
143             deferred.resolve();
144         }, function (error) {
145             logger.error(util.format('Failed to retrieve global session for: :%s',
146                 request.sessionId), error);
147             deferred.reject(error);
148         });
149     }
150 
151     return deferred.promise;
152 
153 };

155 /**
156  * Sets a HTTP only signed cookie on the request.
157  *
158  * @param {Request} request the HTTP request.
159  * @param {Response} response the HTTP response.
160  * @private
161  */
162 Session.prototype._setCookie = function (request, response) {
163     if (!request.globalSession) {
164         return;
165     }
166 
167     var cookie = new Cookie(request, this._cookie),
168         proto = (request.headers['x-forwarded-proto'] || '').toLowerCase(),
169         isSecured = request.connection.encrypted || (proto === 'https');
170 
171     // Browser-session cookies only set-cookie once.
172     if (cookie.expires === null && !request.sessionIsNew) {
173         return;
174     }
175 
176     // Only send secure cookies via https.
177     if (cookie.secure && !isSecured) {
178         return;
179     }
180 
181     var val = 's:' + signature.sign(request.sessionId, config.secret);
182     var cookieValue = cookie.serialize(this._key, val);
183     logger.debug(util.format('Setting cookie: %s', cookieValue));
184     response.setHeader('Set-Cookie', cookieValue);
185 };
186 
187 /**
188  * Saves the session at the end of the request.
189  *
190  * @param {Request} request the HTTP request.
191  * @param {Response} response the HTTP response.
192  * @private
193  */
194 Session.prototype._saveSessionOnEnd = function (request, response) {
195     var end = response.end,
196         self = this;
197 
198     response.end = function (data, encoding) {
199         response.end = end;
200 
201         if (!request.globalSession) {
202             response.end(data, encoding);
203             return;
204         }
205 
206         self._store.save(request.globalSession).then(function () {
207             logger.debug(util.format('Saved global session for: %s', request.sessionId));
208             response.end(data, encoding);
209         }, function (error) {
210             logger.error(util.format('Failed to save global session for: %s', request.sessionId),
211                 error);
212             response.end(data, encoding);
213         });
214     };
215 };
216 
217 /**
218  * Generates a unique random id on 16bytes using crypto module.
219  *
220  * @returns {String} the unique Session id.
221  * @private
222  */
223 Session.prototype._generateSessionId = function () {
224     return crypto.randomBytes(16).toString('hex');
225 };

227 module.exports = Session;
