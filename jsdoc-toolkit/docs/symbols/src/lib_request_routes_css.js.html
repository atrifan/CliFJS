1 "use strict";
  2 
  3 var path = require('path'),
  4     fs = require('fs'),
  5     componentRegistry = require('../component_registry');
  6 
  7 function CssHandler() {}
  8 
  9 CssHandler.prototype.match = function (request) {
 10     var requestPath = request._parsedUrl.pathname,
 11         splitPath = requestPath.split('/').slice(1);
 12     if(splitPath.length >= 3 && splitPath.indexOf('css') !== -1
 13         && path.extname(requestPath) === '.css') {
 14         if(splitPath[0] === 'core') {
 15             request.component = {
 16                 id: 'core',
 17                 cssPath: splitPath,
 18                 useSession: false
 19             }
 20         } else {
 21             request.component = {
 22                 id: splitPath[0],
 23                 cssPath: splitPath,
 24                 version: splitPath[1]
 25             }
 26         }
 27         return true;
 28     } else {
 29         return false;
 30     }
 31 };
 32 
 33 CssHandler.prototype.handle = function (request, response) {
 34     response.setHeader('Content-Type', 'text/css; charset=UTF-8');
 35     response.setHeader('Cache-Control', 'no-cache, must-revalidate');
 36     response.setHeader('Pragma', 'no-cache');
 37     response.setHeader('Expires', new Date(Date.now() - 1000 * 60 * 60).toUTCString());
 38     var componentId = request.component.id,
 39         componentVersion = request.component.version,
 40         cssPath = request.component.cssPath.slice(2).join('/');
 41 
 42     if(componentId === 'core') {
 43         //asta nu merge de regandit
 44         fs.readFile(path.join(process.cwd(), '/components', request._parsedUrl.pathname), function (err, data) {
 45             if(err) {
 46                 throw new Error(err);
 47             }
 48             response.writeHead(200, {
 49                 'Content-Type': 'text/css' });
 50            response.write(data);
 51            response.end();
 52         });
 53     } else if (componentRegistry._componentMap[componentId] &&
 54             componentRegistry._componentMap[componentId].versions.indexOf(componentVersion) !== -1)  {
 55         var location = componentRegistry._componentMap[componentId].configuration[componentVersion].location;
 56 
 57         var location = path.join(location, '/client/', cssPath);
 58 
 59         fs.readFile(location, function (err, data) {
 60             if(err) {
 61                 //daca eroare trebuie inlcuita cumva componenta
 62                 throw new Error(err);
 63             }
 64             response.writeHead(200, {
 65                 'Content-Type': 'text/css' });
 66             response.write(data);
 67             response.end();
 68         });
 69 
 70     }
 71 };
 72 
 73 
 74 module.exports = new CssHandler();
