1 "use strict";
  2 
  3 var config = require('./configuration'),
  4     logger = require('./logging/logger').get(),
  5     path = require('path');
  6 
  7 function FactoryStore() {};
  8 
  9 FactoryStore._instance = null;
 10 
 11 FactoryStore.get = function () {
 12     if(!FactoryStore._instance) {
 13         if(!config.session || !config.session.store) {
 14             logger.info('No implementation for SessionStore available falling to MemoryStore');
 15             FactoryStore._instance = require('./store/memory');
 16         } else {
 17             FactoryStore._instance = require(process.cwd() + path.join(config.session.store));
 18         }
 19     }
 20 
 21     return FactoryStore._instance;
 22 };
 23 
 24 FactoryStore.prototype.createNewSession = function (sessionId) {
 25     logger.fatal('No implementation available for session store `createNewSession`');
 26     return;
 27 };
 28 
 29 FactoryStore.prototype.get = function (sessionId, componentId) {
 30     logger.fatal('No implementation available for session store `createNewSession`');
 31     return;
 32 };
 33 
 34 FactoryStore.prototype.destroy = function (session) {
 35     logger.fatal('No implementation available for session store `createNewSession`');
 36     return;
 37 };
 38 
 39 FactoryStore.prototype.save = function (session) {
 40     logger.fatal('No implementation available for session store `createNewSession`');
 41     return;
 42 };
 43 
 44 module.exports = FactoryStore;
